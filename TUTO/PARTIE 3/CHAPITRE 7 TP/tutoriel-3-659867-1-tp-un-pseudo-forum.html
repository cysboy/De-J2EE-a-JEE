TP : Un pseudo forum

Le tutoriel que vous êtes en train de lire est en bêta-test. Son auteur souhaite que vous lui fassiez part de vos commentaires pour l'aider à l'améliorer avant sa publication officielle. Notez que le contenu n'a pas été validé par l'équipe éditoriale du Site du Zéro.
Par cysboy
Difficulté : Difficile
Avec tout ce que nous avons vu, nous allons faire un TP assez conséquent.
Celui-ci reprendra une bonne partie des précédents chapitres et j'espère qu'il vous plaira.

Je vous préviens tout de suite, ce TP sera très difficile... Il vous demandera beaucoup de réflexion et la gestion de beaucoup de cas différents ! Toujours prêt ?

Cahier des charges

Donc, je vais vous demander de faire un pseudo forum. Pourquoi pseudo, tout simplement car nous n'avons pas encore vu comment interagir avec une base de données... Nous allons donc devoir ruser pour stocker nos topics et nos messages.

Nous allons partir d'une structure de classes très simple :




Maintenant, vous devrez faire toute une série de choses :
  • faire une page qui affiche la liste des forums disponibles avec leurs topics associés, sous forme de tableau;
  • faire de cette page votre page d'accueil.
  • dans chaque forum, mettre un lien qui permette d'y ajouter un nouveau topic;
  • dans chaque forum, afficher la liste des topics ainsi que l'auteur, la date de création et un lien pour aller le visualiser directement;
  • la page qui permet d'ajouter un topic doit afficher la liste des topics existants;
  • faire un formulaire sur la page qui liste les topics d'un forum afin d'en ajouter un nouveau;
  • ajouter un lien sur chaque topic afin de pouvoir visualiser la liste de ses messages;
  • une page qui permet de lister les messages d'un topic avec un formulaire donnant la possibilité d'y ajouter un message;
  • gérer les différentes exceptions que cette application pourrait générer, en redirigeant vers une page spéciale;
  • gérer l'erreur 404;
  • stocker les différents forums, topics et messages dans un SortedMap que vous sérialiserez dans un répertoire de votre choix;
  • le répertoire que vous aurez choisi devra être défini dans un paramètre de contexte et utilisé ainsi;
  • afin de pouvoir récupérer, à chaque requête, la liste des forums, topics et messages, vous devrez faire un filtre qui se chargera d'aller désérialiser votre collection et la mettre dans un attribut du contexte;
  • accéder à votre collection depuis et grâce au contexte de l'application (défini par le filtre);
  • lorsque vous créerez un nouveau topic ou un nouveau message, vous devrez contôler les données saisies par l'utilisateur;
  • gérer les cas où le forum et / ou le topic n'existe pas;
  • lors du premier affichage, la collection d'objet étant inexistante, vous devrez gérer ce cas afin de générer une collection de forums par défaut et sérialiser cette dernière afin de l'utiliser ultérieurement.

Pfiou, ça en fait des choses à gérer...

import java.io.BufferedInputStream;
import java.io.BufferedOutputStream;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.ObjectInputStream;
import java.io.ObjectOutputStream;
import java.util.HashMap;
import java.util.Map;

import javax.servlet.ServletContext;

public class Serializer {

	static ObjectInputStream ois;
    static ObjectOutputStream oos;
	
	public static synchronized void serialize(ServletContext context, Map<String, Forum> listForum) 
			throws IOException,FileNotFoundException{
		//Nous récupérons le chemin de sauvegarde depuis le contexte
		String path = context.getRealPath("/") + context.getInitParameter("pathToSave");
		try {
        	oos = new ObjectOutputStream(
        				new BufferedOutputStream(
        						new FileOutputStream(
        								new File(path))));
        	
        	oos.writeObject(listForum);
        	
        } catch (FileNotFoundException e) {
            e.printStackTrace();
            throw e;
        } 
		finally{
			if(oos != null)
				oos.close();
		}
	}
	
	public static synchronized Map<String, Forum> unSerialize(ServletContext context) 
			throws IOException, ClassNotFoundException{
		//Nous récupérons le chemin de sauvegarde depuis le contexte
		String path = context.getRealPath("/") + context.getInitParameter("pathToSave");
		Map<String, Forum> list = new HashMap<String, Forum>();

        	//On récupère maintenant les données !
        	ois = new ObjectInputStream(
    				new BufferedInputStream(
    						new FileInputStream(
    								new File(path))));
            
			list = (Map<String, Forum>)ois.readObject();

			e.printStackTrace();
			throw e;
		}
		finally{
			if(ois != null)
				ois.close();
		}
		
		return list;
	}

}

package com.sdz.model;
import java.io.Serializable;
import java.util.SortedMap;
import java.util.TreeMap;
import com.sdz.exception.TopicNotFoundException;

public class Forum implements Serializable{

	private String nom;
	private SortedMap<String, Topic> listTopics = new TreeMap<String, Topic>();
	
	public Forum(){}
	public Forum(String nom){this.nom = nom;}
	
	public String getNom() {
		return nom;
	}
	public void setNom(String nom) {
		this.nom = nom;
	}
	public SortedMap<String, Topic> getTopics() {
		return listTopics;
	}
	public void addTopic(Topic top) {
		listTopics.put(top.getTimestamp().toString(), top);
	}
	public Topic getTopic(Object nom) throws TopicNotFoundException{
		//Dans le cas où les forum demandé n'existe pas...
		if(!listTopics.containsKey(nom))
			throw new TopicNotFoundException();
		
		return listTopics.get(nom);
	}
}

package com.sdz.model;

public enum ForumDefinition {
	C("C"), JAVA("JAVA"), PHP("PHP"), HTML("HTML");
	
	private String nom;
	
	private ForumDefinition(String name){
		nom = name;
	}
	public String toString(){
		return nom;
	}
	
	public static boolean exists(String str) {
	    for (ForumDefinition me : ForumDefinition.values()) {
	        if (me.name().equalsIgnoreCase(str))
	            return true;
	    }
	    return false;
	}
}

public class ForumManager {
	
	public static Map<String, Forum> getListeForum(ServletContext context){

package com.sdz.servlets;

import java.io.IOException;
import java.util.Map;

import javax.servlet.ServletException;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

import com.sdz.controler.Controler;
import com.sdz.controler.TopicControler;
import com.sdz.exception.ForumNotFoundException;
import com.sdz.model.Forum;
import com.sdz.model.ForumManager;
import com.sdz.model.Message;
import com.sdz.model.Serializer;
import com.sdz.model.Topic;

public class ListTopic extends HttpServlet {
	
	protected void doGet(HttpServletRequest request, HttpServletResponse response) 
			throws ServletException, IOException {
		Object forum = request.getParameter("forum");
		Map<String, Forum> listForums = (Map<String, Forum>)getServletContext().getAttribute("forums");
		
		if(!listForums.containsKey(forum)){
			throw new ForumNotFoundException();			
		}
		Forum fofo = listForums.get(forum); 
		request.setAttribute("forumActuel", fofo );
		request.getRequestDispatcher("/vue/listTopic.jsp").forward(request, response);
	}
	
	protected void doPost(HttpServletRequest request, HttpServletResponse response) 
			throws ServletException, IOException {

		String auteur = request.getParameter("auteur");
		String titre = request.getParameter("titre");
		String message = request.getParameter("message");
		String forum = request.getParameter("forum");
		
		Map<String, Forum> list = ForumManager.getListeForum(getServletContext());
		Controler controler = new TopicControler();
		//On contrôle que le paramèter envoyé existe dans notre liste de forum
		if(list.containsKey(forum) && controler.control(request, response)){
			
			//On créé le nouveau topic
			Topic topic = new Topic(auteur, titre);
			topic.addMessage(new Message(auteur, message));
			list.get(forum).addTopic(topic);
			Serializer.serialize(getServletContext(), list);
		}
		//On redirige vers la liste des topic
		doGet(request, response);
	}	
}

import java.io.IOException;

import javax.servlet.ServletException;

import com.sdz.model.ForumManager;
import com.sdz.model.Message;
import com.sdz.model.Serializer;
import com.sdz.model.Topic;

public class ListMessage extends HttpServlet {
	protected void doGet(HttpServletRequest request, HttpServletResponse response) 
			throws ServletException, IOException {
		Object forum = request.getParameter("forum");
		Object top = request.getParameter("topic");
				
		Map<String, Forum> listForums = (Map<String, Forum>)getServletContext().getAttribute("forums");
		
		if(!listForums.containsKey(forum)){
			request.getRequestDispatcher("/").forward(request, response);
			return;
		}
		
		Topic topic = new Topic();
		//L'exception qui peu être levée renverra sur la page d'erreur
		topic = listForums.get(forum).getTopic(top);
		
		
		Forum fofo = listForums.get(forum);
		request.setAttribute("forumActuel", fofo );
		request.setAttribute("topicActuel", topic);
		request.getRequestDispatcher("/vue/listMessage.jsp").forward(request, response);
	}
	
	protected void doPost(HttpServletRequest request, HttpServletResponse response) 
			throws ServletException, IOException {

		String auteur = request.getParameter("auteur");
		String message = request.getParameter("message");
		String forum = request.getParameter("forum");
		String top = request.getParameter("topic");
		Topic topic = new Topic();
		
		Map<String, Forum> list = ForumManager.getListeForum(getServletContext());
		Controler controler = new MessageControler();
		//On contrôle que le paramèter envoyé existe dans notre liste de forum
		if(list.containsKey(forum) && controler.control(request, response)){
			//Si le topic n'est pas trouvé, l'exception enverra sur la page d'erreur			
			topic = list.get(forum).getTopic(top);
			topic.addMessage(new Message(auteur, message));
			list.get(forum).addTopic(topic);
			Serializer.serialize(getServletContext(), list);
		}
		//On affiche la page
		doGet(request, response);
	}
}

package com.sdz.filtre;

import java.io.FileNotFoundException;
import java.io.IOException;
import java.util.HashMap;
import java.util.Map;

import javax.servlet.Filter;
import javax.servlet.FilterChain;
import javax.servlet.FilterConfig;
import javax.servlet.ServletException;
import javax.servlet.ServletRequest;
import javax.servlet.ServletResponse;

import com.sdz.model.Forum;
import com.sdz.model.ForumDefinition;
import com.sdz.model.Serializer;

/**
 * Servlet Filter implementation class ForumFilter
 */
public class ForumFilter implements Filter {

    private FilterConfig config;

	public void destroy() { }
	
	/**
	 * Ce filtre va se charger de stocker les objets sérialisés dans
	 * une variable de contexte pour pouvoir l'utiliser dans l'application
	 * et va permettre de la recharger à chaque requête...
	 */
	public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) 
			throws IOException, ServletException {
		
		Map<String, Forum> list = new HashMap<String, Forum>();
		try {
			list = Serializer.unSerialize(config.getServletContext());
			config.getServletContext().setAttribute("forums", list);
		} 
		catch(FileNotFoundException e){ 
			//Dans ce cas, c'est la première utilisation, le fichier n'existe pas...
			//Nous allons donc créer une liste de forum par defaut. :)			
			list.put(ForumDefinition.C.toString(), new Forum(ForumDefinition.C.toString()));
			list.put(ForumDefinition.JAVA.toString(), new Forum(ForumDefinition.JAVA.toString()));
			list.put(ForumDefinition.PHP.toString(), new Forum(ForumDefinition.PHP.toString()));
			list.put(ForumDefinition.HTML.toString(), new Forum(ForumDefinition.HTML.toString()));
			Serializer.serialize(config.getServletContext(), list);
			config.getServletContext().setAttribute("forums", list);
		}
		catch (ClassNotFoundException e) { e.printStackTrace();}
		
		chain.doFilter(request, response);
	}
	
	public void init(FilterConfig fConfig) throws ServletException {
		config = fConfig;
	}
}
